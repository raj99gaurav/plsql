SET SERVEROUTPUT ON;
DECLARE
  v_name VARCHAR2(30);
  --Declare Cursor 
  CURSOR cur_RGVLOGS IS 
  SELECT first_name FROM EMPLOYEES 
  WHERE employee_id < 105;
BEGIN
  OPEN cur_RGVLOGS;
  LOOP 
    FETCH cur_RGVLOGS INTO v_name; 
    DBMS_OUTPUT.PUT_LINE (v_name); 
    EXIT WHEN cur_RGVLOGS%NOTFOUND; 
  END LOOP;--Simple Loop End
  CLOSE cur_RGVLOGS;
END;
